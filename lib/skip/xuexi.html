Linux

常用命令

cat 标准输出或打印
echo 标准输入
clear 清屏

用户操作

useradd 创建新用户
passwa 设置密码
su 切换用户 -l 同时切换环境
groupadd 添加用户组
groupdel 删除用户组
gpasswd 设置群组密码 #可用来将用户添加至组内或将用户移除组 -a 将用户加入到群组 -d 将用户从群组删除 -r 取消群组密码
例： gpasswd -a user
group 目录文件操作
mkdir #创建目录 -p创建目录树 也可同时创建多个目录
touch #创建文件
pwd #显示当前路径
ln -s file1 lnk1 #创建一个指向文件或目录的软链接
ln file1 lnk1 #创建一个指向文件或目录的物理链接
mv dir1 new_dir #重命名/移动 一个目录

cd命令

cd /home 或者 cd ~ #进入当前用户’/home’目录
cd .. #返回上一级目录
cd ../.. #返回上两级目录
cd - #返回上次所在的目录

ls命令

ls #查看目录中的文件 以短格式查看
ls -F #查看目录中的文件
ls -l #显示文件和目录的详细资料 以长格式查看
ls -a #显示隐藏文件
ls *[0-9]* #显示包含数字的文件名和目录名

rm命令

rm -f file1 #删除一个叫做 'file1' 的文件'
rmdir dir1 #删除一个叫做 'dir1' 的目录'
rm -rf dir1 #删除一个叫做 'dir1' 的目录并同时删除其内容
rm -rf dir1 dir2 #同时删除两个目录及它们的内容

cp命令（复制）

cp file1 file2 #复制一个文件
cp dir/* . #复制一个目录下的所有文件到当前工作目录
cp -a /tmp/dir1 . #复制一个目录到当前工作目录
cp -a dir1 dir2 #复制一个目录
